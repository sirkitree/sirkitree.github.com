<script>
(function () {
  var container = document.querySelector('.poem-hang-on');
  if (!container) return;
  if (window.matchMedia && matchMedia('(prefers-reduced-motion: reduce)').matches) return;

  var canvas = container.querySelector('.dust-canvas');
  var ctx = canvas.getContext('2d');
  var dpr = window.devicePixelRatio || 1;
  var particles = [];
  var w = 0, h = 0;
  var rafId = null;
  var rgb = readColor();

  function readColor() {
    var c = getComputedStyle(container).color;
    var m = c.match(/(\d+(?:\.\d+)?)\s*,\s*(\d+(?:\.\d+)?)\s*,\s*(\d+(?:\.\d+)?)/);
    return m ? [parseFloat(m[1]), parseFloat(m[2]), parseFloat(m[3])] : [120, 120, 120];
  }

  function resize() {
    dpr = window.devicePixelRatio || 1;
    var rect = container.getBoundingClientRect();
    w = rect.width;
    h = rect.height;
    canvas.width = Math.max(1, Math.floor(w * dpr));
    canvas.height = Math.max(1, Math.floor(h * dpr));
    canvas.style.width = w + 'px';
    canvas.style.height = h + 'px';
    ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
  }

  function spawn(prefill) {
    var p = {
      x: prefill ? Math.random() * w : -8,
      y: Math.random() * h,
      vx: 0.25 + Math.random() * 0.9,
      vy: (Math.random() - 0.5) * 0.15,
      life: 1,
      decay: 0.0012 + Math.random() * 0.003,
      size: 0.3 + Math.random() * 1.3,
      wob: Math.random() * Math.PI * 2,
      wobS: 0.008 + Math.random() * 0.025
    };
    particles.push(p);
  }

  function tick() {
    ctx.clearRect(0, 0, w, h);
    if (particles.length < 70 && Math.random() < 0.45) spawn(false);

    var alive = [];
    for (var i = 0; i < particles.length; i++) {
      var p = particles[i];
      p.wob += p.wobS;
      p.x += p.vx;
      p.y += p.vy + Math.sin(p.wob) * 0.25;
      p.life -= p.decay;
      if (p.life <= 0 || p.x > w + 20) continue;
      ctx.fillStyle = 'rgba(' + rgb[0] + ',' + rgb[1] + ',' + rgb[2] + ',' + (p.life * 0.22) + ')';
      ctx.beginPath();
      ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
      ctx.fill();
      alive.push(p);
    }
    particles = alive;
    rafId = requestAnimationFrame(tick);
  }

  function start() {
    resize();
    rgb = readColor();
    particles = [];
    for (var i = 0; i < 30; i++) spawn(true);
    cancelAnimationFrame(rafId);
    rafId = requestAnimationFrame(tick);
  }

  var rt;
  window.addEventListener('resize', function () {
    clearTimeout(rt);
    rt = setTimeout(resize, 120);
  });

  document.addEventListener('visibilitychange', function () {
    if (document.hidden) {
      cancelAnimationFrame(rafId);
      rafId = null;
    } else if (!rafId) {
      rafId = requestAnimationFrame(tick);
    }
  });

  var themeObserver = new MutationObserver(function () { rgb = readColor(); });
  themeObserver.observe(document.documentElement, { attributes: true, attributeFilter: ['class', 'data-theme'] });
  themeObserver.observe(document.body, { attributes: true, attributeFilter: ['class', 'data-theme'] });

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', start);
  } else {
    start();
  }
})();
</script>
